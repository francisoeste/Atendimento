CREATE PROCEDURE 'sp_insert_cliente' {
  pCLIENTE varchar(60),
  pCNPJ char(14),
  pINSC_ESTADUAL varchar(9) '',
  pRAZAO varchar(80),
  pFANTASIA varchar(50),
  pENDERECO varchar(50),
  pNUMERO int(10),
  pCOMPLEMENTO varchar(40),
  pBAIRRO varchar(30),
  pCEP char(5),
  pREFERENCIA varchar(50),
  pOBS blob DEFAULT NULL,
  pINSC_MUNICIPAL varchar(11),
  pCPF char(11)'',

}
BEGIN
INSERT INTO cliente ( CLIENTE, CNPJ, INSC_ESTADUA, RAZAO, FANTASIA, ENDERECO, NUMERO, COMPLEMENTO, BAIRRO, CEP, REFERENCIA, OBS, INSC_MUNICIPAL, CPF)
		    VALUES ( pCLIENTE, pCNPJ, pINSC_ESTADUA, pRAZAO, pFANTASIA, pENDERECO, pNUMERO, pCOMPLEMENTO, pBAIRRO, pCEP, pREFERENCIA, pOBS, pINSC_MUNICIPAL, pCPF);

END;



DELIMITER //

BEGIN
	INSERT INTO cliente ( CLIENTE, CNPJ, INSC_ESTADUA, RAZAO, FANTASIA, ENDERECO, NUMERO, COMPLEMENTO, BAIRRO, CEP, REFERENCIA, OBS, INSC_MUNICIPAL, CPF)
		    VALUES ( CLIENTE, CNPJ, INSC_ESTADUA, RAZAO, FANTASIA, ENDERECO, NUMERO, COMPLEMENTO, BAIRRO, CEP, REFERENCIA,OBS, INSC_MUNICIPAL, CPF);
END //

DELIMITER ;
